// define local decimal point character
const decimalPoint = (parseFloat('1.1') === 1.1)? '.' : ',';
const invalidDP = (decimalPoint === '.')? ',': '.';
// change commas to dots for decimal points
function standardizeDP(text) {
    try {
        if (String(text).length > 0) {
            text = Number(text.replace(invalidDP, decimalPoint));
        };
    } catch(e) {
        console.log('number value cannot be evaluated for ', text, '; ', e.message);
    }
    return text;
};
// functions for collecting elements
getQuestionId = function(elt) {
    return elt.closest("[id^='question']").id.match(/\d+/g)[0];
};
getElementsDictByIdPrefixOrClass = function(selector) {
    let all_elements = document.querySelectorAll(`[id^="${selector}"], .${selector}`);
    let elements = {};
    all_elements.forEach(element => {
        let elt_id = getQuestionId(element);
        if (!(elt_id in elements)) {
            elements[elt_id] = [];
        };
        elements[elt_id].push(element);
    });
    return elements;
};
getElementsDictByListOfIdPrefixesOrClasses = function(selectors) {
    let elements = {};
    for (let selector of selectors) {
        let all_elements = document.querySelectorAll(`.${selector}, [id^="${selector}"]`)
        all_elements.forEach(element => {
            let elt_id = getQuestionId(element);
            if (!(elt_id in elements)) {
                elements[elt_id] = [];
            };
            elements[elt_id].push(element);
        });
    }
    return elements;
};

const answerTypes = ['correct_answer', 'wrong_answer', 'no_answer'];
function setAnswerTo(box,answerType) {
    box.classList.remove(...answerTypes.filter(l => l != answerType));
    box.classList.add(answerType);
};
function checkIfAllAnswered(qid, qno) {
    let selector_id = '#question' + qid + ' .correct_answer';
    let inputs_correct = document.querySelectorAll(selector_id);
    let no_correct = inputs_correct.length;
    if (no_correct == qno) {
        // make inputs unresponsive, show success info
        const fontSize = 16;
        for(let box_idx=0; box_idx<inputs_correct.length;box_idx++){
            let box = inputs_correct[box_idx];
            let elt = document.createElement('span');
            elt.classList.add("correct_answer", "replaced");
            elt.style.width = getTextWidthInPx(box.value, fontSize);
            elt.innerHTML = box.value;
            box.parentElement.insertBefore(elt, box);
            box.classList.add('hidden');
        };
        document.getElementById('success_question' + qid).classList.remove('hidden');
    };
};
function checkAnswer(event, i, series_length) {
    let current_target = event.target;
    let qid = getQuestionId(current_target);
    if (event.key === "Enter" || event.key === "Tab" || !event.key) {
        let current_value = current_target.value.replace(/\s+/g, '').toLowerCase();
        let current_value_num = standardizeDP(current_target.value.replace(/\s+/g, '').toLowerCase());
        if ((current_value !== '') && ((current_value_num == table_answers[qid][i])||(current_value == table_answers[qid][i]))) {
            setAnswerTo(current_target, 'correct_answer');
            checkIfAllAnswered(qid, series_length);
        } else if (String(current_value).length > 0) {
            setAnswerTo(current_target, 'wrong_answer');
            checkIfAllAnswered(qid, series_length);                   
        } else {
            setAnswerTo(current_target, 'no_answer');
        };
    } else {
        setAnswerTo(current_target, 'no_answer');
        checkIfAllAnswered(qid, series_length);
    };
};

// add truth / false styles and actions
function swapTrueFalse(event) {
    let elt = event.target;
    let qid = getQuestionId(elt);
    let is_elt_truth = truths[qid].includes(elt);
    let elt_id = is_elt_truth ? truths[qid].indexOf(elt) : falses[qid].indexOf(elt);
    let elt_value = is_elt_truth ? 1 : 0;
    let elt_opposite = is_elt_truth ? falses[qid][elt_id] : truths[qid][elt_id];
    elt.classList.toggle('chosen');
    if (current_choices[qid][elt_id] != -1) {
        elt_opposite.classList.toggle('chosen')
    };
    if (current_choices[qid][elt_id] != elt_value) {
        current_choices[qid][elt_id] = elt_value;
    } else {
        current_choices[qid][elt_id] = 1 - elt_value;
    };
};
for (let qid in truths) {
    let buttons_truth = truths[qid];
    let buttons_false = falses[qid];
    [...buttons_truth,...buttons_false].forEach(elt => {
        elt.addEventListener('click', swapTrueFalse);
    });
};

const all_answers = getElementsDictByIdPrefixOrClass('one_choice');
function getIdsAndGroups(answers) {
    let answers_ids = {};
    let answers_in_groups = {}
    for (let qid in answers) {
        let q_answers = answers[qid];
        parents = [];
        for (let i=0;i<q_answers.length;i++) {
            let ans = q_answers[i];
            if (!parents.includes(ans.parentElement)) {
                parents.push(ans.parentElement);
            };
        }
        let ans_ids = [];
        let groups = {};
        // ans_ids will consist of [parent_id, elt_id], where ..._id are natural numbers
        for (let i=0;i<q_answers.length;i++) {
            let ans = q_answers[i];
            let parent = ans.parentElement;
            let parent_id = parents.indexOf(parent);
            if (parent_id in groups) {
                groups[parent_id].push(ans)
            } else {
                groups[parent_id] = [ans];
            };
            let elt_id = i - parent_id * (q_answers.length / parents.length);
            ans_ids.push([parent_id, elt_id]);
        };
        answers_ids[qid]= ans_ids;
        answers_in_groups[qid] = groups;
    };
    return [answers_ids, answers_in_groups]
};

const answer_ids_and_groups = getIdsAndGroups(all_answers);

        group_elts.forEach(e => e.classList.remove('on'));
        elt.classList.add('on');
        current_choices[qid][group_id] = elt_id + 1;
    } ;
};

function selectTab(event){
    event.preventDefault();
    let tab_clicked = event.currentTarget;
    let result = tab_clicked.id.match(/tab-(\d+)-(\d+)/);
    let section = parseInt(result[1]);
    let position = parseInt(result[2]);
    let section_tabs = document.querySelectorAll(`[id^="tab-${section}-"]`);
    let section_contents = document.querySelectorAll(`[id^="tabcontent-${section}-"]`);

    section_tabs.forEach(tab => {
        tab.classList.remove('chosen');
    });
    section_contents.forEach(content => {
        content.classList.remove('visible');
    });

    tab_clicked.classList.add('chosen');
    document.getElementById(`tabcontent-${section}-${position}`).classList.add('visible');
};

changeMultipleChoice = function(event, i, qid, checkboxes, options, which=0) {
    let current_checkbox = checkboxes[i];
    let current_option = options[i];
    multiple_many_current_answers[qid][which][i] = 1 - multiple_many_current_answers[qid][which][i];
    if ((event.target == current_checkbox && current_checkbox.checked) || (event.target != current_checkbox && !current_checkbox.checked)) {
        current_option.classList.add('chosen');
        current_checkbox.checked = true;
    } else {
        current_option.classList.remove('chosen');
        current_checkbox.checked = false;
    };
};
for (let qid in multiple_choice_checkboxes) {
    let checkboxes = multiple_choice_checkboxes[qid];
    let options = multiple_choice_option_fields[qid];
    let confirm_button = multiple_choice_confirm_buttons[qid][0];
    let feedback_fields = multiple_choice_feedback_fields[qid];
    confirm_button.addEventListener('click', (event) => {
        feedback_fields.forEach(feedback_field => {
            feedback_field.classList.add('hidden')
        });
        let correct = true;
        let correct_not_chosen = 0;
        for (let i=0; i<multiple_many_current_answers[qid][0].length; i++) {
            let current_answer = multiple_many_current_answers[qid][0][i];
            let current_correct_answer = multiple_many_correct_answers[qid][0][i];
            correct = correct && (current_answer == current_correct_answer);
            if (current_answer != current_correct_answer) {
                if(feedback_fields[i].innerHTML){
                    feedback_fields[i].classList.remove('hidden')
                };
            } else if (current_answer < current_correct_answer) {
                correct_not_chosen ++;
            };
        };
        if (correct) {
            feedback_fields.forEach(feedback_field => {
                feedback_field.classList.add('hidden')
            });
            confirm_button.style.display = 'none';
            document.querySelector(`#question${qid} .failure`).classList.add('hidden');
            document.querySelector(`#question${qid} .success`).classList.remove('hidden');
            checkboxes.forEach(elt => {elt.style.display = 'none'});
            for (let j=0; j<options.length; j++) {
                let new_elt = options[j].cloneNode(true);
                options[j].parentNode.replaceChild(new_elt, options[j]);
            }
        } else if (correct_not_chosen > 0) {
            document.querySelector(`#question${qid} .failure`).classList.remove('hidden');
        };
        if (correct_not_chosen == 0) {
            document.querySelector(`#question${qid} .failure`).classList.add('hidden');
        }
    });
    
    for (let i=0; i<checkboxes.length; i++) {
        checkboxes[i].addEventListener('change', e => changeMultipleChoice(e, i, qid, checkboxes, options));
        options[i].addEventListener('click', e => changeMultipleChoice(e, i, qid, checkboxes, options));
    };
};

function getTextWidth(input_str, fontSize=16) {
    text = document.createElement("span");
    document.body.appendChild(text);

    text.style.font = "times new roman";
    text.style.fontSize = fontSize + "px";
    text.style.height = 'auto';
    text.style.width = 'auto';
    text.style.position = 'absolute';
    text.style.whiteSpace = 'no-wrap';
    text.innerHTML = input_str;

    width = Math.ceil(text.clientWidth);
    document.body.removeChild(text);
    return width;
};

function getTextWidthInPx(input_str, fontSize=16) {
    width = getTextWidth(input_str, fontSize)
    formattedWidth = `${width + 10}px`;
    return formattedWidth;
};

function getSlidePath(previous_path, idx) {
    let idx_end = previous_path.match(/.+\d+/)[0].length;

cumulative_positions = function(freq_list, value) {
    let current_sum = 0;
    for (let idx=0;idx<freq_list.length;idx++) {
        current_sum += freq_list[idx];
        if (value < current_sum) {
            return [idx, value - current_sum + freq_list[idx]]
        };
    };
    return [-1, -1]
};

function moreLessGuessed(event) {

            elt.innerHTML = current_target.value;
        } else if (String(current_value).length > 0) {
